1 /**
  2  * Special Javascript file which gets inserted between JQuery and JQuery mobile scripts when bundling.
  3  * <p>
  4  * The problem is that jQuery Mobile does implicit init and tries to render the page instantly
  5  * the script is loaded.
  6  * <p>
  7  * jQuery Mobile relies and jQuery event mechanism and we must set some events for jQuery Mobile
  8  * to run our own init code after jQuery is loaded. This is what this file is for.
  9  * <p>
 10  *
 11  *
 12  */
 13 
 14 mobilize.extend(mobilize, 
 15 		
 16  	/** @lends mobilize */
 17 	{
 18 	
 19 	/**
 20 	 * Mobilize code which can be run only after loading jQuery.
 21 	 */
 22 	onJQueryLoad : function() {
 23 	
 24 		if(!jQuery) {
 25 			throw "Need to have jQuery loaded before entering here."
 26 		}
 27 	
 28 		mobilize.bindInitEvents();
 29 	},
 30 	
 31 	/**
 32 	 * Must be called before template loading,
 33 	 * as immediately when jQuery Mobile script tag is inserted to DOM,
 34 	 * some of its event handlers are run.
 35 	 */
 36 	bindInitEvents : function() {
 37 		
 38 		// Assign jQuery Mobile event start up handler 
 39         //mobilize.bind(window.document, "mobileinit", mobilize.onMobileInit);
 40 	
 41         $(window.document).bind("mobileinit", mobilize.onMobileInit);
 42 	},
 43 	
 44 	/**
 45 	 * jQuery mobile initializer handler 
 46 	 * 
 47 	 * @param {Object} e
 48 	 */
 49 	onMobileInit : function(e) {
 50 				
 51 		// We'll manage our own workflow and don't want to jQuery Mobile
 52 		// start doing things instantly when the script is loaded
 53 		mobilize.log("Disabling jQuery Mobile autoInitialize, was:" + $.mobile.autoInitialize);
 54 		$.mobile.autoInitialize = false;
 55 		$.mobile.ajaxEnabled = false;
 56 		
 57 		mobilize.jQueryMobileLoaded = true;
 58 		mobilize.prepareFinish();
 59 		
 60 	},
 61 	
 62 });
 63 mobilize.onJQueryLoad();
