1 /**
  2  * Wordpress CMS mobilization using mobilization.js.
  3  * <p>
  4  * This functionality is retrofitted to core <i>mobilize</i> class.
  5  * <p>
  6  * @namespace Wordpress mobilization
  7  * <p>
  8  * @extends mobilize
  9  */
 10 var mobilizeWordpress = {
 11 	
 12 	/**
 13 	 * Wordpress specific default options setter.
 14 	 * <p>
 15 	 * This sets default download locations for files using
 16 	 * minified bundles and CDN URI layout. Override
 17 	 * this in init() call if you want to do something else.
 18 	 * <p>
 19 	 * @see mobilize.init
 20 	 * 
 21 	 */
 22 	initPlugins : function() {			   
 23 	
 24 	   mobilize.extend(mobilize.cdnOptions, {
 25 		   bundleName : "mobilize.wordpress",
 26 		   cssBundles : ["css/mobilize.wordpress.mobile.min.css"],
 27 		   javascriptBundles : ["js/mobilize.wordpress.mobile.min.js"],
 28 		   template : "templates/wordpress.html"
 29 	   });
 30 	   
 31 	},
 32 	
 33 	constructBody : function() {
 34 		
 35 		mobilize.log("Wordpress constructBody()");
 36 				
 37 		if($("body").hasClass("single-post")) {
 38 			// Post type page
 39 			this.constructPostPage();
 40 		} else {
 41 			// Front page
 42 			this.constructFrontPage();
 43 		}
 44 		this.constructHeader();
 45 		this.constructFooter();
 46 		
 47 	},
 48 	
 49 	constructHeader : function() {
 50         // Set page heading from <title> tag
 51 		
 52 		var header;
 53         header = $("#mobile-body div[data-role=header]");
 54 		
 55         var title = $("head title").text();
 56         header.append("<h1>" + title + "</h1>");	
 57 		
 58 		mobilize.constructBackButton(header);
 59 	},
 60 	
 61 	constructFooter : function() {
 62 	    //$("#mobile-body div[data-role=footer]").append($("#site-info"));     	
 63 		
 64 		// Put site slogan to footer 
 65 		$("#mobile-body div[data-role=footer]").append($("#site-description"));         
 66 	},
 67 
 68     /**
 69      * Create back button
 70      * 
 71      * Point to Home if not already there
 72      */	
 73 	constructBackButton : function(header) {
 74 		
 75 		if (!$("body").hasClass("home")) {
 76 			header.prepend("<a data-icon=home href='/'>Home</a>");
 77 		}
 78 	},
 79 	
 80 	constructPostPage : function(){
 81 		
 82 		// Add back button to header
 83 		// The header is defined in to template.html(core.html)
 84 		var header;
 85 		header = $("#mobile-body div[data-role=header]");
 86 								
 87 		var content = $("#mobile-body div[data-role=content]");
 88 		var entry_content = $(".entry-content");
 89 		content.append(entry_content);
 90 		
 91 		// Add comment area which can be hidden.
 92 		
 93 		// jQuery element which controls the collapsiple section
 94 		var collapsible = $('<div id="comment-collapsible" data-role="collapsible" data-collapsed="true">');
 95 		collapsible.appendTo(content);
 96 				
 97 		var header = $('<h3>');
 98 		// TODO: Get from page for localization
 99 		header.text("Comments");
100 		header.appendTo(collapsible);
101 		
102 		var comments = $("<p>");
103 		comments.append($("#comments"));
104 		comments.appendTo(collapsible);
105 		
106 	},
107 	
108 	/**
109 	 * This is a nasty one
110 	 */
111 	constructFrontPage : function() {
112 	
113 		var baseurl = mobilize.baseurl(window.location.href);
114 	    
115 		// Move box on the left hand to body first
116 		var content = $("#mobile-body div[data-role=content]");
117 		if(content.size() == 0) {
118 			throw "No template content section to fill in";
119 		}
120 		
121 		content.append($("#current"));
122 		
123 		var entries = $(".post");
124 		function outputter(list, input, a) {
125 			
126 			var output = $("<li role='option'>");
127 			var title = input.find(".entry-title");
128 			
129 			var text = title.text();
130 			var tmp = $("<h3 class='ui-li-heading'>");
131 			
132 			var link = title.find("a");
133             if(link) 
134             {
135                 var href = link.attr("href");
136                 // Add mobilize=true to get the new page show mobile version
137             	// This is mainly for testing on pc as the mobile detection 
138 				// should handle this automatically when the page is loaded.
139                 if(href && mobilize.getUrlVars().mobilize !== undefined) 
140                 {
141                     if(href.indexOf("http://") >= 0) 
142                     {
143                         var newurl = mobilize.addUrlVar(href, "mobilize=" + mobilize.getUrlVars().mobilize);
144                         link.attr("href", newurl);
145                     }
146                 }
147                 link.text(text);
148                 tmp.append(link);
149             }
150             title = $("<div class='ui-btn-text'>");
151 			
152 			var date = input.find(".entry-date").text();
153 			var info = $('<p class="ui-li-aside">');
154 			info.text(date);
155 			tmp.append(info);
156 			
157 			title.append(tmp);
158 			output.append(title);
159 			
160 			var entry_content = input.find(".entry-content");
161 			output.append(entry_content);
162 			
163 			output.appendTo(list);
164 
165 		}
166 
167 		var mainNavigation = mobilize.createNavigationBox(entries, "Recent headlines", outputter);
168 		content.append(mainNavigation);		
169 		
170 		// Pages
171         var menu = $(".menu");
172         var items = menu.find("li");
173         list = $("<ul data-inset='true' data-role='listview'>");
174         list.prepend("<li data-role='list-divider'>Pages</li>"); // TODO: Localization
175         
176         items.each(function(){
177             var output = $("<li role='option'>");
178             output.append($(this).find("a"));
179             output.appendTo(list);
180         });
181         content.append(list);
182         
183 	},
184 	
185 	/**
186 	 * This is called when jQuery Mobile internal transform is done.
187 	 * 
188 	 * We can start binding jQuery Mobile UI elements
189 	 * @param {Object} event
190 	 * @param {Object} data
191 	 */
192 	 bindEventHandlers : function(event, data) {
193 	 	
194 		// XXX: Something is wrong with $ shortcut in this point
195 		// jQuery() event bindings work, but not when using $
196 		
197 		mobilize.log("Installing Wordpress event handlers");
198 	    var collapsible = jQuery("#comment-collapsible");
199 
200 		mobilize.log("Found collapsible:" + collapsible.size());
201 		collapsible.bind("expand", mobilize.onCommentsOpen);	    
202      },
203 	
204 	/**
205 	 * Special handler which will move focus to comments when comment button is pressed
206 	 */
207 	onCommentsOpen : function(event, data) {
208 
209 	   	mobilize.log("comments open");
210         
211 		var x = 0;
212 		var y = event.target.offsetTop;
213 		window.scrollTo(x, y);
214         mobilize.log(event);
215 	}
216 };

218 mobilize.extend(mobilize, mobilizeWordpress);
