246 	initCloud : function() {
247 		
248 		var opts = mobilize.cdnOptions;
249 		
250 		if(!opts.cloud) {
251 			// Cloud services have been disabled
252 			return;
253 		}
254

256 		if(!opts.cloudBaseURL) {
257 			// Try to extract cloud URL from our <script> tag
258 			var scripts = document.getElementsByTagName("script");
259 			var src = null;
260 			for(var i=0; i<scripts.length; i++) {
261 				var script = scripts[i];
262 				var klass = script.getAttribute("class");
263 				
264 				if(klass == "mobilize-js-source") {
265 					
266 					// Found our script tag
267 					src = script.getAttribute("src");
268 					
269 					if(!src) {
270 						// Inline script tag
271 						continue;
272 					}
273 					
274 					var base = mobilize.baseurl(src);
275 					
276 					mobilize.log("Found script source URL" + base);
277 					
278 					// Remove /js/ from the end of the URL
279 					base = base.substring(0, base.length-4);
280 					
281 					opts.baseURL = base;		
282 					
283 					break;
284 				}
285 			}

365 	/** 
366 	 * <p>Get baseurl from url by ignoring file and url parameters</p> 
367 	 * 
368 	 * <b>Example</b>
369 	 * <pre>
370 	 * mobilize.baseurl(window.location.href)
371 	 * </pre>
372 	 * @param url : Url to parse
373 	 * 
374 	 * */
375 	baseurl : function (aUrl) {
376 		
377 		var end;
378 		var url;
379 		
380 		end = aUrl.indexOf('?');
381 		
382 		if(end <= 0) {
383 			end = aUrl.length-1;
384 		}
385 		
386 		url = aUrl.slice(0, end);
387 		// Ignore slash at the end of url
388 		if(url[url.length-1] == "/" ) {
389 			url = url.slice(0,url.length-2);
390 		}
391 		
392 		// But add the slash to result for convenient concat
393 		end = url.lastIndexOf("/") + 1;
394 		url = url.slice(0,end);
395 		
396 		return url;
397 	},

412 	addUrlVar : function(aURL, aNewVar){
413 	    var args = mobilize.getUrlVars(aURL);
414 	    var newurl = aURL.split("?",1)[0];
415 	    newurl += "?";
416 	    
417 	    var items = []
418 	    for(var i = 0; i < args.length; i++) {
419 	        var a = args[i];
420 	        var value = args[a];
421 	        items.push(a + "=" + value);
422 	    }
423 	    
424 	    items.push(aNewVar);
425 	    
426 	    newurl += items.join("&");
427 	    return newurl;
428 	},
429 	/** 
430 	 * Read URL parameters to dict.
431 	 * 
432 	 * See: http://jquery-howto.blogspot.com/2009/09/get-url-parameters-values-with-jquery.html
433 	 */
434 	getUrlVars : function (aURL)
435 	{
436 	    if(!this._urlvars) {
437 	        this._urlvars = {};
438 	    }
439 	    if(!aURL) {
440             aURL = window.location.href;
441         }
442 	    
443 		// Cache window.location.href call results
444 		if(this._urlvars[aURL]) {
445 			return this._urlvars[aURL];
446 		}
447 		
448 	    var vars = [], hash;
449 	    mobilize.log("aURL:"+aURL)
450         
451 	    if(aURL.indexOf("#") >= 0 ){
452 	        aURL = aURL.slice(0,aURL.indexOf("#"));
453 	        mobilize.log("aURL:"+aURL)
454 	    }
455 	    var hashes = aURL.slice(aURL.indexOf('?') + 1).split('&');
456 	    
457 	    for(var i = 0; i < hashes.length; i++)
458 	    {
459 	        hash = hashes[i].split('=');
460 	        vars.push(hash[0]);
461 	        vars[hash[0]] = hash[1];
462 	    }
463 	    
464 	    this._urlvars[aURL] = vars;
465 	    return vars;
466 	},

628 	/**
629 	 * Load JS and CSS files needed on the mobile page.
630 	 * <p>
631 	 * This function provides some logic for caching 
632 	 * the result, so that the files are not reloaded again.
633 	 * <p>
634 	 * Note that Javascript is loaded synchronously
635 	 * and transform() won't proceed until Javascript
636 	 * (jQuery) is completely loaded.
637 	 * <p>
638 	 * CSS are loaded asyncrhonously.
639 	 */
640 	loadMobileResources : function() {		
641 		var i;
642 		var self = this;
643 		var cdn = mobilize.cdnOptions;
644 		
645 		var jsCompleteCount = 0;
646 		
647 		mobilize.log("Constructing mobile <head>");
648 		
649 		function onJSComplete() {
650 			jsCompleteCount++;
651 			
652 			if(jsCompleteCount == 1) {
653 				// Assume jQuery loaded
654 			}
655 			
656 			// Proceed to tempalte transform
657 			if(jsCompleteCount >= cdn.javascriptBundles.length) {							
658 				self.loadMobileTemplate();
659 			}						
660 		}
661 				
662 		for(i=0; i<cdn.javascriptBundles.length; i++) {
663 			var js = cdn.javascriptBundles[i];
664 			mobilize.log("Loading js:" + js);
665 			mobilize.loadScript(mobilize.toFullCDNURL(js), onJSComplete);
666 		}

1230 /** Returns a function, which calls the original function after given delay
1231  * with parameters given to the returned function.
1232  * 
1233  * @param {Function} func
1234  * @param {Object} after
1235  * @param {Array}  opts     List of parameters
1236  * Example:
1237  
1238 mobilize.delayedFunction( function(arg1,arg2){
1239         alert("1000="+(arg1 + arg2));
1240     }, 
1241     1000
1242 )(400,600);

1245 mobilize.delayedFunction = function callAfter(func, after, opts){
1246     return function(){        
1247         // Call in the same scope as the original
1248         var scope = mobilize.getattr(opts, "scope", func);        
1249         var args  = arguments;
1250         setTimeout(function(){ func.apply(scope, args);}, after);
1251     };

1256  * @param {Function} func

1258  *

1268 /** Utility for calling functions. 
1269  * Supports callbacks, timeouts, scope change and repeating.
1270  * 
1271  * @param {Function} func Function to call
1272  * @param {Object}   opts  Options:
1273  *                    args  - arguments for the function. Use list for multiple.
1274  *                    scope - Object which is referred to by 'this' in function.
1275  *                    after - Call asynchronously after time in ms
1276  *                    callback - Called after asynchonous operation completes.
1277  *                    repeat   - How many times to repeat the function
1278  */
1279 mobilize.call = function call(func, opts ){
1280     
1281     var scope = mobilize.getattr( opts, "scope", func );
1282     var args  = mobilize.getattr( opts, "args", undefined );
1283     var after = mobilize.getattr( opts, "after", undefined);
1284     var repeat   = mobilize.getattr( opts, "repeat", 1);
1285     var callback = mobilize.getattr( opts, "callback", undefined);
1286     
1287     // Convert single argument to list
1288     if( args ){
1289         if( typeof args != "object" || args.length === undefined ){
1290             args = [args];
1291         }
1292     }
1293 
1294     
1295     if( after !== undefined ){
1296         func = mobilize.scope( func, scope );
1297         
1298         // Call asynchronously
1299         var called = mobilize.scope( function(){
1300                 var result = mobilize.trap(this.func, { args : this.args });
1301                 if( this.callback ){ 
1302                     mobilize.trap(this.callback, {args : result});
1303                 }
1304             },  
1305             {func:func, callback : callback, args : args }
1306         );
1307         
1308         for( var i = 1; i <= repeat; i++){
1309             mobilize.delayedFunction(called, after * i )();
1310         }
1311         
1312     }
1313     else {
1314         for( var i = 0; i < repeat; i++){
1315             var result = func.apply( scope, args);
1316         }
1317     }
1318     
1319     return result;
1320 };
1321 /** Call function and catch exceptions. Logs trace (if supported)
1322  * @param func: Function to call
1323  * @param ots: Options passed to mobilize.call
1324  *             onerror : callback to call instead of logging trace
1325  *             onfinally : Always called with or without exception after 
1326  *                         the function completes.
1327  * */
1328 mobilize.trap = function trap(func, opts){
1329     
1330     var result = undefined;
1331     try{
1332         result = mobilize.call( func, opts );
1333     }catch(e){        
1334         if( opts && opts.onerror ){
1335             opts.onerror(e);
1336         }else{
1337             mobilize.log( "mobilize.call options:" + String(opts) );
1338             mobilize.log( "Called function:"  + func );
1339             if( !e.stack){                                
1340                 mobilize.log( e.sourceURL + ":"+e.line + "\n" + e.name + ":" + e.message );
1341             }
1342             else{           
1343                 mobilize.log( e.stack);
1344             }
1345             // Pass the error on
1346             throw e;
1347         }
1348     }
1349     finally{
1350         if( opts && opts.onfinally ){
1351             opts.onfinally();
1352         }
1353     }
1354     return result;
1355 };

1357 /** Like mobilize.trap, but does not call the function. 
1358  * Returns a decorated function instead.
1359  * 
1360  * Useful for callbacks, which might otherwise fail silently.
1361  * @param func: Function to decorate
1362  * @param options: See options of mobilize.trap and mobilize.call
1363  *                 options.args are overridden with arguments given to decorator
1364  *                 if any given.
1365  */
1366 mobilize.trapped = function trapped(func, options ){
1367     
1368     return function(){
1369         if( !options ){
1370             options = {};
1371         }
1372         
1373         if( arguments.length > 0 ){
1374             options.args = arguments;
1375         }
1376         
1377         if( !options.scope ){
1378             options.scope = func;
1379         }
1380         return mobilize.trap( func, options );
1381     };
1382 };
